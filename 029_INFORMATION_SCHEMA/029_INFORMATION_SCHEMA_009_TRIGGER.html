SQL Server 


select 
  tri.name  AS  TriggerName,
  tri.is_disabled AS IsDisable,
  tri.is_instead_of_trigger AS IsInsteadOfTrigger,
  c.text AS CreateTriggerSQL
from 
  sys.triggers tri LEFT OUTER JOIN
     dbo.syscomments c ON tri.object_id = c.id

PostgreSQL 


SELECT
  tr.tgname AS "触发器名",
  t.relname  AS  "被触发表名",
  p.proname  AS  "触发器函数"
FROM
  pg_trigger tr
    JOIN pg_class t
	  ON (t.oid = tr.tgrelid)
    JOIN pg_proc p
	  ON (tr.tgfoid = p.oid);
