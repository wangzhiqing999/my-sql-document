字符函数

转换为 Unicode 编码字符 TO_NCHAR

Oracle TO_NCHAR 用于将 varchar 之类的字符， 转换为 varchar 类型. 



SELECT 
	TO_NCHAR('A') A , 
	NULLIF(N'A', TO_NCHAR('A')) AA,
	NULLIF('A', TO_NCHAR('A')) AAA	
FROM dual;

 NULLIF('A', TO_NCHAR('A')) AAA
             *
ERROR 位于第 4 行:
ORA-12704: 字符集不匹配





SELECT 
	TO_NCHAR('A') A , 
	NULLIF(N'A', TO_NCHAR('A')) AA
FROM dual;

A  AA
-- --
A
