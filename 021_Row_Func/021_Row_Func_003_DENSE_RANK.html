SELECT
  DENSE_RANK() OVER( ORDER BY SUM(grade) DESC) AS NO,
  student_name,
  SUM(grade) AS AllGrade
FROM
  test_course
GROUP BY
  student_name
ORDER BY
  SUM(grade) DESC

MySQL 


MySQL 不直接支持 DENSE_RANK 函数。
需要用替代方法。

SELECT
  CASE 
    WHEN @AllGrade = AllGrade THEN @rank
    ELSE @rank:= @rank + 1  
  END AS RANK,
  @rownum:=@rownum+1 AS NO,
  student_name,
  @AllGrade:=AllGrade AS AllGrade
FROM
  (SELECT @rownum:=0) rn,
  (SELECT @rank:=1) rk,
  (SELECT @AllGrade:=0) ag,
  (SELECT
    student_name,
    SUM(grade) AS AllGrade
  FROM
    test_course
  GROUP BY
    student_name
  ) SubQuery
ORDER BY
  AllGrade DESC;

SELECT
  CASE 
    WHEN @cn != course_name THEN @rank:= 1 
    WHEN @grade = grade THEN @rank
    ELSE @rank:= @rank + 1  
    END AS RANK,
  CASE 
    WHEN @cn != course_name THEN @rownum:= 1 
    ELSE @rownum:= @rownum + 1  
    END AS No,
  student_name,
  @cn := course_name AS course_name,
  @grade := grade AS grade
FROM
  (SELECT @rownum:=0) r,
  (SELECT @rank:=1) rk,
  (SELECT @cn:='') p,
  (SELECT @grade:=0) g,
  test_course
ORDER BY
  course_name,
  grade DESC;
