SELECT
  utl_raw.bit_and( '0F' , '8') AS "0x0f位与0x08",
  utl_raw.bit_or('2' , '8') AS  "0x02位或0x08",
  utl_raw.bit_xor('0F' , '3') AS "0x0f位异或0x03"
FROM
  dual;

SQL Server 


SELECT
  15 & 8 AS [0x0f位与0x08],
  2 | 8 AS  [0x02位或0x08],
  15 ^ 3 AS [0x0f位异或0x03]

0x0f位与x08  0x02位或x08  0x0f位异或x03
----------- ----------- -----------
8           10          12
