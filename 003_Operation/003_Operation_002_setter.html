Oracle 


两种方式
:=  (普通设置) 与 SELECT  INTO (通过查询设置)

DECLARE
	vi	INT;
BEGIN
	vi := 100;
	dbms_output.put_line(TO_CHAR(vi));	
	SELECT 1024  INTO  vi FROM dual;
	dbms_output.put_line(TO_CHAR(vi));
END;

100
1024

两种方式
SET (普通设置) 与 SELECT (通过查询设置)

DECLARE @i  INT;
SET @i = 100;
PRINT @i;

--	这里仅仅用于演示，就不FROM具体表了.
SELECT @i = 1024;
PRINT @i;
GO

1024
