一元运算符

Oracle 


SQL> SELECT
  2    + 32 AS  "正32",
  3    - 32 AS  "负32",
  4    utl_raw.bit_complement  ('10') AS  "位非32"
  5  FROM
  6    dual;

      正32       负32
---------- ----------
位非32
--------------------------------------------------
        32        -32
EF
DB2
SQL Server 


SELECT
  + 32 AS  [正32],
  - 32 AS  [负32],
  ~ 32 AS  [位非32]

正32         负32         位非32
----------- ----------- -----------
32          -32         -33
(1 行受影响)
