创建角色

Oracle 


SQL> create role test_role;
角色已创建


SQL> grant create any table,create procedure to test_role;
授权成功。




COLUMN "ROLE" FORMAT A25

-- 查询当前数据库的名称为 TEST_ROLE 的角色.
SELECT 
  *
FROM
  dba_roles
WHERE
  role = 'TEST_ROLE';
  

ROLE                      PASSWORD_REQUIRE
------------------------- ----------------
TEST_ROLE                 NO




COLUMN "GRANTEE" FORMAT A16
COLUMN "GRANTED_ROLE" FORMAT A16


-- 查询当前数据库的 TEST_ROLE 角色，都被赋予了哪些用户.
SELECT
  *
FROM
  DBA_ROLE_PRIVS
WHERE
  GRANTED_ROLE = 'TEST_ROLE';

GRANTEE          GRANTED_ROLE     ADMIN_ DEFAUL
---------------- ---------------- ------ ------
TEST             TEST_ROLE        YES    YES

（注：这里的 test 用户， 是创建这个角色的用户）






COLUMN "GRANTEE" FORMAT A16
COLUMN "USERNAME" FORMAT A16
COLUMN "PRIVILEGE" FORMAT A24


-- 查询当前数据库的 TEST_ROLE 角色，都被赋予了哪些系统权限.
SELECT
  *
FROM
  DBA_SYS_PRIVS
WHERE
  GRANTEE = 'TEST_ROLE';


GRANTEE          PRIVILEGE                ADMIN_
---------------- ------------------------ ------
TEST_ROLE        CREATE PROCEDURE         NO
TEST_ROLE        CREATE ANY TABLE         NO
