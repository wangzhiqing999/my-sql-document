触发器

触发器的临时禁用

Oracle 


-- 首先查询触发器信息
SELECT
  trigger_name,
  status
FROM
  user_triggers
WHERE
  table_name='TEST_TRIGGER_TABLE';

TRIGGER_NAME                                                 STATUS
------------------------------------------------------------ ----------------
AFTERDELETETEST                                              ENABLED


-- 临时禁用触发器
ALTER TRIGGER AFTERDELETETEST  DISABLE;


-- 再次查询触发器信息
TRIGGER_NAME                                                 STATUS
------------------------------------------------------------ ----------------
AFTERDELETETEST                                              DISABLED


-- 恢复触发器
ALTER TRIGGER AFTERDELETETEST  ENABLE;


-- 再次查询触发器信息
TRIGGER_NAME                                                 STATUS
------------------------------------------------------------ ----------------
AFTERDELETETEST                                              ENABLED
