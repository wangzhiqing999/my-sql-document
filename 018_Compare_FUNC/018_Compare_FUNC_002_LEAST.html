一般比较函数

取得最小值 LEAST

Oracle 


SQL> SELECT 
  2    LEAST(1,3,5,7,9) A
  3  FROM
  4    dual;

         A
----------
         1

SQL> select least('aa', 'bb', 'cc') A
  2  from dual;

A
--
Aa
DB2 


db2 => SELECT
db2 (cont.) =>   LEAST(1,3,5,7,9) A
db2 (cont.) => FROM SYSIBM.SYSDUMMY1;

A
-----------
          1

  1 条记录已选择。
SQL Server 


-- 比较难受的写法.
SELECT
	MIN(value)
FROM
	( 
		SELECT 1  AS  value	UNION  ALL
		SELECT 3  AS  value	UNION  ALL
		SELECT 5  AS  value	UNION  ALL
		SELECT 7  AS  value	UNION  ALL
		SELECT 9  AS  value
	) subQuery
GO
